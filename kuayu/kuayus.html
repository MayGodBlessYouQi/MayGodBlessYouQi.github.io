# 跨域的方法

  1. webpack配置跨域 创建vue.config.js文件
    vue.config.js文件内容
module.exports = {
  devServer: {
	  //这里的地址是你配置的后端接口
    proxy: 'http://localhost:3000',
  },
};

在src的api文件中配置一个index.js文件

import axios from 'axios';
// 响应拦截器
// 公共请求地址
axios.defaults.baseURL = 'http://localhost:3000';
axios.interceptors.response.use(res => res.data, err => Promise.reject(err));
// 轮播图请求
const getBaner = () => axios.get('/banner');
// 首页列表请求hl
// 2 getHl执行 axios.get 就执行了 拿到结果 这个结果就是getHl的结果
const getHl = () => axios.get('/hl');
export { getBaner,getHl };
export default {};

在需要值得组件上接收一下这个跨域的json

<template>
  <div class="home">
    <div v-if="loading">
      <div class="loadEffect">
      </div>
    </div>
    <div v-else>
    </div>
  </div>
</template>
<script>
// 把方法引入
import { getBaner,getHl } from '../api'
export default {
  name: 'home',
  async created() {
    this.imgs = await getBaner();
    this.list = await getHl();
  },
  data() {
    return {
      loading: true,
      imgs: [],
      list: [],
    }
  }
}
  1. cors 实现跨域 app.js文件添加方法
const app = require('express')();
const banner = require('./banner.js');
// cors
const cors = require('koa-cors');
app.use(cors());

const {
    rfile
} = require('./ff.js');
app.get('/banner', (req, res) => {
    res.json(banner);
})
app.listen(3030, () => {
    console.log('服务器启动成功')
});
  1. Node Express 解决请求跨域请求 index.js文件
import axios from 'axios';

//添加一个请求地址,不需要再额外添加一个vue.config.js文件就可以实现跨域
axios.defaults.baseURL = 'http://localhost:3030';

axios.interceptors.response.use(res => res.data, err => Promise.reject(err));
let getBaner = () => axios.get('/banner');
export {
    getBaner
};
export default {};

app.js文件

const app = require('express')();
const banner = require('./banner.js');
// cors
app.use((req, res, next) => {
  // 配置cors跨域头
  // 允许哪个域名跨域
  res.header('Access-Control-Allow-Origin', '*');
  // 允许什么方法进行跨域
  res.header('Access-Control-Allow-Methods', 'GET,POST,DELETE,PUT');
  next(); // use里面如果不调用next，代码就不继续执行了
})
const {
    rfile
} = require('./ff.js');
app.get('/banner', (req, res) => {
    res.json(banner);
})
app.listen(3030, () => {
    console.log('服务器启动成功')
})
